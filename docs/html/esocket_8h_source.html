115  operator EString() const
116  {
117  Char buf[INET6_ADDRSTRLEN];
118 
119  if (m_addr.ss_family == AF_INET)
120  {
121  if (!inet_ntop(m_addr.ss_family,&((struct sockaddr_in*)&m_addr)->sin_addr.s_addr,buf,sizeof(buf)))
122  throw AddressError_ConvertingToString();
123  }
124  else // AF_INET6
125  {
126  if (!inet_ntop(m_addr.ss_family,&((struct sockaddr_in6*)&m_addr)->sin6_addr.s6_addr,buf,sizeof(buf)))
127  throw AddressError_ConvertingToString();
128  }
129  return EString(buf);
130  }
131 
134  operator UShort() const

136  if (m_addr.ss_family == AF_INET)

138  if (m_addr.ss_family == AF_INET6)
139  return ntohs(((struct sockaddr_in6*)&m_addr)->sin6_port);
140  throw AddressError_UndefinedFamily();

215  Address &setAddress(cpStr addr, UShort port)
216  {
217  clear();
218  if (inet_pton(AF_INET,addr,&((struct sockaddr_in*)&m_addr)->sin_addr) == 1)
219  {
220  ((struct sockaddr_in*)&m_addr)->sin_family = AF_INET;
221  ((struct sockaddr_in*)&m_addr)->sin_port = htons(port);
222  return *this;
223  }
224 
225  clear();
226  if (inet_pton(AF_INET6,addr,&((struct sockaddr_in6*)&m_addr)->sin6_addr) == 1)
227  {
228  ((struct sockaddr_in6*)&m_addr)->sin6_family = AF_INET6;
229  ((struct sockaddr_in6*)&m_addr)->sin6_port = htons(port);
230  ((struct sockaddr_in6*)&m_addr)->sin6_flowinfo = 0;
231  ((struct sockaddr_in6*)&m_addr)->sin6_scope_id = 0;
232  return *this;
233  }
234 
235  throw AddressError_UnknownAddressType();
236  }
