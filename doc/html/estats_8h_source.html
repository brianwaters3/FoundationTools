31 //
32 // According to IANA, the current diameter application ID range is from 0 to 16777361.
33 // Diameter application ID's start at 0 and appear to be "reserved" through 16777215 (0x00FFFFFF).
34 // 3GPP diameter applications ID's appear to have the low order bit of the high order word set to 1 (16777216 decimal or 0x1000000 hex).
35 //

38 #define PFCP_BASE (0xfd000000)
