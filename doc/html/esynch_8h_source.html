105 class EMutexLock
106 {
107 public:
108  EMutexLock(EMutexData &mtx, Bool acquire = True)
109  : m_acquire(acquire), m_mtx(mtx)
110  {
111  if (m_acquire)
112  m_mtx.enter();
113  }
114 
115  ~EMutexLock()
116  {
117  if (m_acquire)
118  m_mtx.leave();
119  }
120 
121  Bool acquire(Bool wait = True)
122  {
123  if (!m_acquire)
124  m_acquire = m_mtx.enter(wait);
125  return m_acquire;
126  }
127 
128 private:
129  EMutexData &m_mtx;
130  Bool m_acquire;
131 
132  EMutexLock();
